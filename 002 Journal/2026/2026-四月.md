### 💪 本月生活平衡状态

```dataviewjs
const currentMonth = moment("2026-04-01", "YYYY-MM-DD");
const startWeek = currentMonth.startOf('month').format('YYYY-[W]ww');
const endWeek = currentMonth.endOf('month').format('YYYY-[W]ww');

const weeks = dv.pages('#week')
  .where(w => w.file.name >= startWeek && w.file.name <= endWeek);

if (weeks.length === 0) {
  dv.paragraph("暂无本月周数据");
} else {
  const fields = ['Health', 'Wealth', 'Relationship', 'Spirituality'];
  const labels = ['💚 健康', '💰 财富', '❤️ 关系', '✨ 精神'];
  
  const stats = fields.map(field => {
    // 收集数字类型的值
    const values = [];
    for (const w of weeks) {
      const v = w[field];
      if (typeof v === 'number' && v > 0) {
        values.push(v);
      }
    }
    const avg = values.length > 0 ? Math.round(values.reduce((a, b) => a + b, 0) / values.length) : 0;
    const status = avg >= 7 ? '🟢 良好' : avg >= 4 ? '🟡 一般' : '🔴 需关注';
    return [labels[fields.indexOf(field)], avg + '/10', status];
  });
  
  dv.table(["维度", "本月平均", "状态"], stats);
  dv.paragraph(`*基于 ${weeks.length} 周的数据汇总*`);
}
```
